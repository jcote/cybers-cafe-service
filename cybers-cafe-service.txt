put this in: /etc/systemd/system/cybers-cafe-service.service

[Unit]
Description=Cybers Cafe socket service daemon
After=sql-proxy.service syslog.target network.target auditd.service
[Service]
ExecStart=/usr/bin/node /home/latte/Code/cybers-cafe-service/app.js
Restart=always
StandardOutput=syslog
StandardError=syslog
SyslogIdentifier=cybers-cafe-service
User=cybers
Group=cybers
Environment=NODE_ENV=production
[Install]
WantedBy=multi-user.target
